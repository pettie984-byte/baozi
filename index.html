// 🟢 處理網頁載入時的 GET 請求：回傳開團狀態與商品列表 function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. 取得「開團參數」工作表 var paramSheet = ss.getSheetByName("開團參數"); var groupName = "我的團購"; var startDateStr = ""; var endDateStr = ""; if (paramSheet) { groupName = paramSheet.getRange("B1").getValue(); // 假設 B1 是團購名稱 startDateStr = paramSheet.getRange("B2").getValue(); // 假設 B2 是開團日期 endDateStr = paramSheet.getRange("B3").getValue(); // 假設 B3 是截止日期 } // 2. 取得「商品清單」工作表 var prodSheet = ss.getSheetByName("商品清單"); var products = []; if (prodSheet) { var prodData = prodSheet.getDataRange().getValues(); // 假設第一行是標題：[商品名稱, 規格, 價格] for (var i = 1; i < prodData.length; i++) { if (prodData[i][0]) { // 確保名稱不為空 products.push({ name: prodData[i][0], spec: prodData[i][1] || "", price: Number(prodData[i][2]) || 0 }); } } } // 3. 計算目前開團狀態 var now = new Date(); var startDate = new Date(startDateStr); var endDate = new Date(endDateStr); var isActive = false; var statusText = "尚未開始"; if (now >= startDate && now <= endDate) { isActive = true; statusText = "進行中"; } else if (now > endDate) { isActive = false; statusText = "已結束"; } var resultData = { result: "success", groupName: groupName, startDate: Utilities.formatDate(startDate, "GMT+8", "yyyy-MM-dd HH:mm"), endDate: Utilities.formatDate(endDate, "GMT+8", "yyyy-MM-dd HH:mm"), isActive: isActive, status: statusText, products: products }; return ContentService.createTextOutput(JSON.stringify(resultData)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ result: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // 🔴 處理前端送出訂單時的 POST 請求 function doPost(e) { try { // 1. 讀取前端以 application/x-www-form-urlencoded 格式傳入的參數 var name = e.parameter.name; var orderString = e.parameter.order; if (!name || !orderString) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": "寫入失敗：遺失姓名或品項資料！" })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } var orderArray = JSON.parse(orderString); // 解析出如 [0, 2, 1, 0] 的數量陣列 // 2. 開啟或建立「訂單紀錄」工作表 var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("訂單紀錄"); if (!sheet) { sheet = ss.insertSheet("訂單紀錄"); // 若是新建立的，自動補上基本標題列 sheet.appendRow(["下單時間", "訂購人姓名"]); } // 3. 準備寫入的一列資料：[當前時間, 訂購人姓名, 數量1, 數量2, 數量3...] var timestamp = new Date(); var rowData = [timestamp, name]; // 將數量陣列串接至 rowData for (var i = 0; i < orderArray.length; i++) { rowData.push(orderArray[i]); } // 4. 追加寫入最後一行 sheet.appendRow(rowData); // 5. 回傳成功狀態與 Headers，以避免跨域問題 return ContentService.createTextOutput(JSON.stringify({ "result": "success", "message": "資料順利寫入訂單紀錄！" })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": "後端錯誤: " + error.toString() })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } }
